## 排序

### 初级排序算法

1.深入学习这些相对简单的算法的原因是：1.通过它们熟悉一些术语和简单的技巧；2.简单的算法有时比复杂的算法更加高效；3.有助于改进复杂算法的效率。

2.排序算法的类模板：

```js
class Example {
  constructor(list) {
    this.list = list;
  }

  sort() {
    // 排序算法
  }

  less(v, w) {
    return v < w;
  }

  exch(a, i, j) {
    const t = a[i];
    a[i] = a[j];
    a[j] = a[i];
  }

  show(a) {
    for (let i = 0; i < a.length; i++) {
      console.log('i', a[i]);
    }
  }

  isSorted(a) {
    for (let i = 0; i < (a.length - 1); i++) {
      if (a[i] > a[i + 1]) return false;
    }
    return true;
  }

  test() {
    this.list = [2, 7, 5, 2, 7, 4, 89, 3];
    this.sort(this.list);
    console.log('isSorted', this.isSorted(this.list))
    console.log('this.list', this.list);
  }
}
```

3.选择排序：

```js
class Example {
  constructor(list) {
    this.list = list;
  }

  sort(a) {
    // 选择排序
    for (let i = 0; i < a.length; i++) {
      let min = i;

      for (let j = i + 1; j < a.length; j++) {
        if (this.less(a[j], a[min])) min = j;
      }

      this.exch(a, i, min);
    }
  }

  less(v, w) {
    return v < w;
  }

  exch(a, i, j) {
    const t = a[i];
    a[i] = a[j];
    a[j] = t;
  }

  show(a) {
    console.log(this.isSorted(a), a);
  }

  isSorted(a) {
    for (let i = 0; i < (a.length - 1); i++) {
      if (a[i] > a[i + 1]) return false;
    }

    return true;
  }

  test() {
    this.list = [2, 7, 5, 2, 7, 4, 89, 3];
    this.sort(this.list);
    this.show(this.list);
  }
}
```
